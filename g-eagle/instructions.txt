      ↓
 Supervisor Agent  ←──────────────────────────────┐

      ↓                                            │

      ↓                                            │
 SQL Generator ──→ Executor ──→ [Error?] ──→ SQL Fixer (x N retries)

              (Schema / Logic / Data / Business)
                      ↓

                      ↓

                      ↓

### `llm/llm_factory.py`

python

`from langchain_openai import ChatOpenAI
from langchain_anthropic import ChatAnthropic
from langchain_google_genai import ChatGoogleGenerativeAI

def get_llm(config: dict):
    provider = config["llm"]["provider"]
    model = config["llm"]["model"]
    temp = config["llm"].get("temperature", 0.0)

    if provider == "openai":
        return ChatOpenAI(model=model, temperature=temp)
    elif provider == "anthropic":
        return ChatAnthropic(model=model, temperature=temp)
    elif provider == "gemini":
        return ChatGoogleGenerativeAI(model=model, temperature=temp)
    else:
        raise ValueError(f"Unsupported LLM provider: {provider}")`

### `db/db_factory.py`

python

`from abc import ABC, abstractmethod

class BaseExecutor(ABC):
    @abstractmethod
    def execute(self, sql: str) -> dict:
        pass

class AthenaExecutor(BaseExecutor):
    def __init__(self, config):
        import boto3
        self.client = boto3.client("athena", region_name=config["region"])
        self.s3_output = config["s3_output"]
        self.database = config["database"]

    def execute(self, sql: str) -> dict:
        # Athena async execution with polling
        response = self.client.start_query_execution(
            QueryString=sql,
            QueryExecutionContext={"Database": self.database},
            ResultConfiguration={"OutputLocation": self.s3_output}
        )
        query_id = response["QueryExecutionId"]
        # poll until complete, return results
        return self._poll_and_fetch(query_id)

    def _poll_and_fetch(self, query_id):
        import time
        while True:
            status = self.client.get_query_execution(
                QueryExecutionId=query_id
            )["QueryExecution"]["Status"]["State"]
            if status in ["SUCCEEDED", "FAILED", "CANCELLED"]:
                break
            time.sleep(1)
        if status != "SUCCEEDED":
            raise Exception(f"Query {status}")
        return self.client.get_query_results(QueryExecutionId=query_id)

def get_executor(config):
    provider = config["database"]["provider"]
    if provider == "athena":
        return AthenaExecutor(config["database"]["athena"])
    # extend: snowflake, postgres, bigquery
    raise ValueError(f"Unsupported DB: {provider}")`

### `agents/discovery/glue_connector.py`

python

`import boto3

class GlueConnector:
    def __init__(self, region="us-east-1"):
        self.client = boto3.client("glue", region_name=region)

    def search(self, query: str) -> list:
        """Return tables with metadata as list of dicts."""
        response = self.client.search_tables(SearchText=query, MaxResults=10)
        results = []
        for table in response.get("TableList", []):
            results.append({
                "source": "glue",
                "name": table["Name"],
                "database": table["DatabaseName"],
                "columns": [
                    {"name": c["Name"], "type": c["Type"]}
                    for c in table.get("StorageDescriptor", {})
                                   .get("Columns", [])
                ],
                "description": table.get("Description", "")
            })
        return results`

### `agents/discovery/pinecone_connector.py`

python

`from pinecone import Pinecone
from langchain_openai import OpenAIEmbeddings

class PineconeConnector:
    def __init__(self, config):
        pc = Pinecone(api_key=config["PINECONE_API_KEY"])
        self.index = pc.Index(config["pinecone"]["index_name"])
        self.embedder = OpenAIEmbeddings(
            model=config["pinecone"]["embedding_model"]
        )

    def search(self, query: str, top_k=5) -> list:
        vector = self.embedder.embed_query(query)
        results = self.index.query(vector=vector, top_k=top_k,
                                   include_metadata=True)
        return [
            {
                "source": "pinecone",
                "name": m["metadata"]["table_name"],
                "columns": m["metadata"]["columns"],
                "description": m["metadata"].get("description", ""),
                "score": m["score"]
            }
            for m in results["matches"]
        ]`

### Retry Logic in LangGraph (edge function)

python

`def should_retry(state: AgentState) -> str:
    config = load_config()
    max_attempts = config["retry"]["max_attempts"]

    if state["execution_error"] is None:
        return "validate"                    # success → go to validation
    elif state["retry_count"] < max_attempts:
        return "fix_sql"                     # still retries left → fix
    else:
        return "human_fallback"              # exhausted → escalate`

## 🔗 Phase 10 — Full LangGraph Graph

python

`# graph/graph_builder.py

from langgraph.graph import StateGraph, END
from langgraph.checkpoint.sqlite import SqliteSaver

def build_graph(config):
    graph = StateGraph(AgentState)

    # Add nodes
    graph.add_node("supervisor",        supervisor_node)
    graph.add_node("discovery",         discovery_node)
    graph.add_node("schema_retriever",  schema_retriever_node)
    graph.add_node("query_planner",     query_planner_node)
    graph.add_node("sql_generator",     sql_generator_node)
    graph.add_node("executor",          executor_node)
    graph.add_node("sql_fixer",         sql_fixer_node)
    graph.add_node("validator",         validation_council_node)
    graph.add_node("summarizer",        summarizer_node)
    graph.add_node("human_handoff",     human_handoff_node)
    graph.add_node("memory_writer",     memory_writer_node)

    # Entry point
    graph.set_entry_point("supervisor")

    # Linear happy path
    graph.add_edge("supervisor",       "discovery")
    graph.add_edge("discovery",        "schema_retriever")
    graph.add_edge("schema_retriever", "query_planner")
    graph.add_edge("query_planner",    "sql_generator")
    graph.add_edge("sql_generator",    "executor")

    # Retry loop
    graph.add_conditional_edges("executor", should_retry, {
        "validate":       "validator",
        "fix_sql":        "sql_fixer",
        "human_fallback": "human_handoff"
    })
    graph.add_edge("sql_fixer", "executor")

    # Post-validation routing
    graph.add_conditional_edges("validator", after_validation, {
        "summarize":      "summarizer",
        "retry":          "sql_generator",
        "escalate":       "human_handoff"
    })

    graph.add_edge("summarizer",     "memory_writer")
    graph.add_edge("human_handoff",  "memory_writer")
    graph.add_edge("memory_writer",  END)

    # Enable checkpointing (session memory)
    memory = SqliteSaver.from_conn_string("checkpoints.db")
    return graph.compile(checkpointer=memory,
                         interrupt_before=["human_handoff"])`
